PROGRAM
VARIABLES
  result: int;
  num: int

SUBPROGRAMS  
	SUBPROGRAM factorial(n: int, & res: int ) 
	VARIABLES
	   res2: int
	{
	   IF (n == 0): {res = 1} []
	      (true): {factorial(n-1,res2); res = n * res2}
	   ENDIF
	}
 
# Programa Principal 
{
  read num;
  factorial(num, result);
  write result
}
